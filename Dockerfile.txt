# Dockerfile
FROM python:3.11-slim

# System libs needed by OpenCV/scikit-image
RUN apt-get update && apt-get install -y --no-install-recommends \
    libglib2.0-0 libgl1 && \
    rm -rf /var/lib/apt/lists/*

# Work in /app/web so your ../SavedFiles path resolves
WORKDIR /app/web

# First copy only requirements to leverage Docker cache
COPY requirements.txt /app/requirements.txt
RUN pip install --no-cache-dir -r /app/requirements.txt

# Copy everything (both web/ and SavedFiles/)
COPY . /app

# HF Spaces expects port 7860
EXPOSE 7860

# Run your Flask app with gunicorn (no edits to app.py)
# app module is 'app' because WORKDIR is /app/web
CMD ["gunicorn", "-w", "2", "-k", "gthread", "-b", "0.0.0.0:7860", "app:app"]
